--lua script
--镇地守卫
--[[键入'ss'并按分号即可按模板生成脚本]]--
--[[当前脚本内的全局变量]]--

thisNPC = nil
InitFnTable = {}
FinaFnTable = {}
MainFnTable = {}

--[[如果此NPC包含标准任务，请将下面一行的最前面的'-'去掉！]]

--#include "data\language\LangCode.txt" once
--#include "data\functions\Activity\SnowArea\SecretArea.txt" once --秘境传送员


--[[此处为NPC对白]]
NpcDialogue = {  
   { say = Lang.NormalTalk.h00129, },

}

--[[脚本主函数]]--
function main(sysarg)

  local rand = System.getRandomNumber( table.getn(NpcDialogue) )
  local str = NpcDialogue[rand +1].say

  for i = 1, table.getn(MainFnTable) do
    str = str .. MainFnTable[i](sysarg)
  end

  return str
end

--[[初始化函数]]--
function initialization(npcobj)
  local i
  thisNPC = npcobj
  --SPID = TSystem.getGlobalVariant( "spid" )  
  for i = 1, table.getn(InitFnTable) do
    InitFnTable[i]( npcobj )
  end
end

--[[析构化函数]]--
function finalization(npcobj)
  local i
  for i = 1, table.getn(FinaFnTable) do
    FinaFnTable[i]( npcobj )
  end
  thisNPC = nil
end
