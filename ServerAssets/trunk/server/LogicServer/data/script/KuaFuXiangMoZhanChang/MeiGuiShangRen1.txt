--[[此处为NPC对白]]
NpcDialogue = {
   { say = Lang.NormalTalk.n02078, },

}

--[[脚本主函数]]--
function main(sysarg)
   local str = ""
	local dialog_count = table.getn(NpcDialogue)
	if dialog_count > 0 then
		local rand = System.getRandomNumber(dialog_count)
		str = NpcDialogue[rand +1].say
	end
	for k, func in ipairs(MainFnTable) do
		str = str .. func(sysarg)
	end
	return str.."\\<#BL"..Lang.Activity.a00032.."/@exit>"  --<离开>
   
end




--[[注销任务的系统回调函数]]


--[[初始化函数]]--
function initialization(npcobj)
  local i
  thisNPC = npcobj
  for i = 1, table.getn(InitFnTable) do
    InitFnTable[i]( npcobj )
  end
end

--[[析构化函数]]--
function finalization(npcobj)
  local i
  for i = 1, table.getn(FinaFnTable) do
    FinaFnTable[i]( npcobj )
  end
  thisNPC = nil
end
