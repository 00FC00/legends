--主动怪的AI的样本配置
{
    
    aiType=3, --ai类型,1表示被动ai,2表示主动ai, 3表示固定路线寻路怪(有固定路径移动), 4表示乱跑怪,5表示中立怪，6表示NPC，7表示宠物AI,8表示英雄ai
    
    --怪物的技能列表，分别表示技能的ID和等级
    skills=
    {
       --id表示技能id,level表示技能等级,autoCD 表示是否自动进入技能CD。如果配置为true，该技能会在进入战斗之后马上进入CD。
       -- 主要用于避免高级技能在一开始就使用出来。
       --event 表示在什么事件下释放，
	   --默认是0可以不配置,1表示死亡的时候触发,2表示刷出来的时候触发,3表示空闲的时候使用(比如嘲讽技能),4被攻击时触发 5非死亡复活技能触发 6表示踩踏触发(暂时只对图腾怪有效)
       
       {id=111,level=1,event=0}, 

    },
    
    
    patrolRadius =0, --运动半径,将在这个半径里动来动去。对于逃跑怪，用这个设置每次逃跑的运动半径 **如果怪物为不可移动怪 则此项必须配置为 0 
    
    pursuitDistance=20, --追击范围，与目标的距离超过这个距离就回归

    watchDistance =4,  --可视范围，主动怪进入这个返回就将攻击,被动怪可以不配置
    
    leftHomeMaxDistance =20, --离开巢位的最大的距离，一旦离开出生点超过这个距离就回归
 
    returnHomeRenewRate =100,  --怪物回归的时候，将按什么比例恢复自己的HP,MP,这里填百分数，100就是100%,0就是0%
	
	returnHomeDelToxic = false,	-- = 怪物回归的时候，是否驱散毒BUFF，true:驱散 默认不驱散
   
    stopMinTime =10,      --怪物停留的最小时间，单位是秒，怪物计算停留时间的时候是在 stopMinTime和stopMaxTime随机选一个

    stopMaxTime =20,      --怪物停留的最大时间，单位是秒

    moveStopInterval = 2000,  --移动之间的停留，单位ms
	
    
    checkenemyInterval=3000, -- 主动怪战斗中检测敌人间隔时间
    optionals={},       --针对一些特殊的ai使用key,value的格式配置,到时候扩展
	
	movePath = {10,11,12,13,14,15},	--怪物移动路径点序列 每2个数字当一个坐标点组X,Y （必须配置2的倍数个数字，且aiType必须配置为3护送怪）
    
	
    ------------下面是宠物ai的配置，怪物的可以不配置
    transferDis =20, -- 离开主人的距离，超过这个将自动传送到主人的身边

    battleFollowDis =6,      --离开主人超过这个距离，将向玩家移动
   
    followMasterTime =3000, 	--宠物离开战斗后多少时间以后将往主人方向走，单位ms
	
	
	hitSetTargetRate =30, --被攻击的时候，设置攻击自己的实体为目标概率，0-100有效,0表示不切换目标，100表示必定切换目标
	
},
