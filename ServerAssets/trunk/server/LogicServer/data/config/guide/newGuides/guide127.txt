			condList =

			action = { type = 1, helpId = 8, data = 0, desc = Lang.GuideLang.desc18 },

    --第2步 指向鉴定导航按钮
		{
			condList = 
			{
				{ type = 7,  helpId = 0, dataList = {130,} },    --任务进行中
				{ type = 1,  helpId = 120, dataList = {0,} },      --锻铸界面打开

			},

			action = { type = 2, helpId = 121, data = 3, desc = Lang.GuideLang.desc23 },
		},

		{
			condList = 
			{
				{ type = 7,  helpId = 0, dataList = {130,} },    --任务进行中

			},

      --引导玩家点击物品导航栏 的 背包物品
			action = { type = 2, helpId = 122, data = 2, desc = Lang.GuideLang.desc20 },
		},

    --第4步 判断身上是否有穿好装备（有就直接点击物品了）
		{
			condList = 
			{
				{ type = 7,  helpId = 0, dataList = {130,} },    --任务进行中
				{ type = 1,  helpId = 120, dataList = {0,} },      --锻铸界面打开
				{ type = 3,  helpId = 121, dataList = {3,} },      --导航选中第3项
				{ type = 3,  helpId = 122, dataList = {1,} },      --物品导航栏选中第一项
				{ type = 21,  helpId = 0, dataList = {69,80,91,} },    --身上有这些物品  **
				{ type = 10,  helpId = 128, dataList = {0,} },      --鉴定cell没数据
			},

      --引导玩家直接点击 身上装备 的物品
			action = { type = 10, helpId = 123, data = {69,80,91,}, desc = Lang.GuideLang.desc24 },
		},

     --第5步 指向背包物品中的装备
		{
			condList = 
			{
				{ type = 7,  helpId = 0, dataList = {130,} },    --任务进行中
				{ type = 1,  helpId = 120, dataList = {0,} },    --锻铸界面打开
				{ type = 3,  helpId = 121, dataList = {3,} },    --导航选中第3项
				{ type = 3,  helpId = 122, dataList = {2,} },    --物品导航栏选中第二项
				{ type = 13,  helpId = 120, dataList = {69,80,91,} },  --背包有这些物品
				{ type = 10,  helpId = 128, dataList = {0,} },   --鉴定cell没数据
			},

      --引导玩家直接点击 背包物品 的装备
			action = { type = 10, helpId = 124, data = {69,80,91,}, desc = Lang.GuideLang.desc24 },
		},

    --第6步 点鉴定按钮
		{
      condList = 
			{
				{ type = 7,  helpId = 0, dataList = {130,} },   --任务进行中
				{ type = 1,  helpId = 120, dataList = {0,} },   --锻铸界面打开

      condList =

				{ type = 2,  helpId = 120, dataList = {0,} },    --强化窗口不可见
				{ type = 6,  helpId = 0, dataList = {130,} },    --不存在任务
      },
      
      --无内容
      action = { type = 0, helpId = 0, data = 0, desc = Lang.GuideLang.desc9 },
    },
	},
},
