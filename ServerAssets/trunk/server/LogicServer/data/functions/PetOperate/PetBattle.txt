--宠物出动，完成任务
function onPetBattle( sysarg, args )
	--完成相关的任务
	for i = 1, #PetBattlequestData do
		local qData = PetBattlequestData[i]
		if qData then
			Actor.setQuestValue(sysarg, qData.qid, qData.tid, qData.count)
		end
	end
end
