function CR(sysarg,rankName)
	if not rankName then return end
	local ranking = Ranking.getRanking(rankName)
	if ranking then
		Ranking.clearRanking(ranking)
		Actor.sendTipmsg(sysarg, "clear ranking success!", ttFlyTip)
	end
end
