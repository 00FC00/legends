CombatOpFuncList = {}

EventCallDispatcher.HandlerCombatOp = function (sysarg, args)
  --print("战力参数："..args[1])
  if args[1] == nil then return end   
  local func = CombatOpFuncList[args[1]]
  if func ~= nil then
    func(sysarg, args) 
  end
end

	AddChanllageTimesOp( sysarg, args )
