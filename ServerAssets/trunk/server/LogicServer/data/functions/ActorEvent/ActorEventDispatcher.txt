--param eventId:事件id
--param args:对应的参数列表
function OnActorEvent(sysarg, eventId, ...)
  --print("eventID:"..eventId)
  local ret = false
  local fnTable = EventCallDispatcher[eventId]
  if fnTable then
	  for i = 1, #fnTable do
	    ret = fnTable[i](sysarg, arg)
	  end
  end
  return ret
end
