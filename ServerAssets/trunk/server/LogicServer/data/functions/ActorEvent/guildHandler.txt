EventCallDispatcher.HandlerGuildOp = function (sysarg, args)
 --print("参数："..args[1])
  if args[1] == nil then return end   --第1个参数  升级帮派技能
  local func = GuildOpFuncList[args[1]]
  if func ~= nil then
    func(sysarg, args) 
  end
end
