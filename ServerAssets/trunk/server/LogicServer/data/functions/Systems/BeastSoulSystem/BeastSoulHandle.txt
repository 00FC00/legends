BeastSoulOpFuncList = {}

EventCallDispatcher.HandlerBeastSoulOp = function (sysarg, args) 
	--print("≤Œ ˝£∫"..args[1])
	if args[1] == nil then return end
	local func = BeastSoulOpFuncList[args[1]]
	if func ~= nil then
		func(sysarg, args) 
	end
end

--«Î«Ûº§ªÓ¡È ﬁ

	ActBeast(sysarg,args)
end

end

--¡È ﬁ∏ΩªÍ

	AttachBeast(sysarg,args)
end
