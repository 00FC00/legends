function SendVIPUserTip(sysarg, vipType)
	vipType = vipType or 1
	local tips = {
		Lang.Misc.m000439,
		Lang.Misc.m000440,
		Lang.Misc.m000441,
	}
	
	local tip = tips[vipType]
	local info = string.format(tip, Actor.getName(sysarg))
	System.broadcastTipmsg(info, ttScreenCenter + ttChatWindow)
end
